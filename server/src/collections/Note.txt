hooks: {
        afterChange: [
            async ({ existingItem, updatedItem, req }) => {
                const action = existingItem ? 'update' : 'create';

                await addToChangelog({
                    collection: 'categories',
                    itemId: updatedItem._id,
                    action,
                    userId: req.user._id,
                });
            },
        ],
        afterDelete: [
            async ({ item, req }) => {
                await addToChangelog({
                    collection: 'categories',
                    itemId: item._id,
                    action: 'delete',
                    userId: req.user._id,
                });
            },
        ],
    },
